      document.getElementById('tab-eks').classList.toggle('active', id === 'eks');
      document.getElementById('tab-inc').classList.toggle('active', id === 'inc');
      document.getElementById('tab-eks').setAttribute('aria-selected', id === 'eks');

      // tokoh aksi FAB: scroll to panel & focus jumlah
      document.querySelector('.panel').scrollIntoView({behavior:'smooth', block:'center'});

      const data = {

        tipe: document.getElementById('tab-eks').classList.contains('active') ? 'pengeluaran' : 'pemasukan'
      };
      // TODO: kirim ke backend atau localStorage

      // document.getElementById('transaksiForm').reset();

      document.getElementById('ket').value = '';
    }
